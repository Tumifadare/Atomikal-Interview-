1    import requests
2    from bs4 import BeautifulSoup
3    from collections import Counter
4    import re
5    import pandas as pd
6    
7    url = 'https://www.cementospacasmayo.com.pe/sostenibilidad/lineamientos'
8    response = requests.get(url)
9    soup = BeautifulSoup(response.text, 'html.parser')
10   
11   text = soup.get_text()
12   
13   words = re.findall(r'\w+', text.lower())
14   word_count = Counter(words)
15   
16   df = pd.DataFrame(word_count.items(), columns=['Word', 'Frequency'])
17   df.to_excel('WordFreqListAtomikal.xlsx', index=False)
18   print("Word frequency list has been saved to 'WordFreqListAtomikal'.")
